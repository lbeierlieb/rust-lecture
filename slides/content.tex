\begin{frame}[fragile]
    \frametitle{Blocks}
    Blocks are not regularly used on their own, but they are crucial for many other language constructs.
    \pause
    \vfill
    \begin{minted}[linenos]{rust}
fn main() {
    let x = 3;
    {
        let x = {};
    }
    println!("{x}");
}
    \end{minted}
    \vfill
    All \texttt{\{\ldots\}} (except \texttt{\{x\}} inside the string) are blocks.
\end{frame}

\begin{frame}[fragile]
    \frametitle{Blocks}
    \begin{minted}[linenos]{rust}
fn main() {
    let x = 3;
    {
        let x = 5;
        println!("x = {x}");
    }
    println!("x = {x}");
}
    \end{minted}
    \pause
    \vfill
    Output:
    \begin{minted}{text}
x = 5
x = 3
    \end{minted}
    \pause
    \vfill
    \begin{itemize}
        \item Bindings go out of scope at the end of the block they are declared in.
        \item Shadowed bindings can become available again.
    \end{itemize}
\end{frame}

\begin{frame}[fragile]

\begin{frame}[fragile]

    } else {
        b
    }
}

\end{frame}

\begin{frame}[fragile]
    \frametitle{Tuples}
    \begin{itemize}
        \item We now look into more complex types.
        \item A tuple is a fixed-size list of heterogenoeous values.
    \end{itemize}
    \pause
    \vfill
    \begin{minted}[linenos]{rust}
fn main() {
    let a: (u64, char, f32) = (32, 'f', 12.47);
    let b = (13, true, 15);  // type inference works for tuples
    let c: u32 = (5);  // (5) is not a tuple
    let d: (u32,) = (5,);  // one-sized tuples look a bit odd
    let e = ();  // the unit type is just a zero-sized tuple!
}
    \end{minted}
    \pause
    \vfill
    Tuples are convenient to return multiple values from a function.
\end{frame}

\begin{frame}[fragile]
    \frametitle{Advanced Printing}
    \begin{itemize}
        \item Tuples have no default formatter (we noticed already when trying to print the unit value).
        \item Tuples support debug formatting.
    \end{itemize}
    \pause
    \vfill
    \begin{minted}[linenos]{rust}
fn main() {
    println!("{:?}", ());

    \begin{minted}{text}
()
(32, 'f', 12.47)
    \end{minted}
\end{frame}

\begin{frame}[fragile]
    \frametitle{Advanced Printing}
    \begin{itemize}
        \item The \texttt{:} specifies that formatting options follow.
        \item The \texttt{?} specifies debug formatting.
        \item There are a lot more formatting options (\href{https://doc.rust-lang.org/std/fmt/}{fmt documentation}).

    \pause

    \begin{minted}{text}

1000010
fe
    \end{minted}

\begin{frame}[fragile]
    \frametitle{Examples: Arrays, Tuples, Functions, \texttt{while}}
    \begin{minted}[linenos]{rust}
fn min_and_max(array: [u64; 10]) -> (u64, u64) {
    let mut min = u64::MIN;
    let mut max = u64::MAX;
    let mut i = 0;
    while i < 10 {
        let val = array[i];
        if val < min {
            min = val;
        }
        if val > max {
            max = val;
        }
        i += 1;
    }
    (min, max)
}
    \end{minted}

\begin{frame}[fragile]
    \frametitle{Structs}
    If a \texttt{struct} with unnamed fields behaves like a tuple but is more verbose, what's the point?
    \pause
    \vfill
    \begin{minted}[linenos]{rust}
struct Point(f64, f64);  // (x, y), point in 2d space
struct Direction(f64, f64);  // (x, y), direction/vector in 2d space
struct Range(f64, f64);  // (min, max), a range in 1d space

    \begin{itemize}
        \item The types store the same data.
        \pause
        \vfill
        \item Using \texttt{struct}s instead of tuples lets the compiler check that you do not accidently pass a \texttt{Direction} to where a \texttt{Point} is expected.
    \end{itemize}

\begin{frame}[fragile]
    \frametitle{\texttt{match} Expressions}
    To cover multiple cases, the \texttt{match} expression is more comfortable than a chain of \texttt{if let}s.
    \pause
    \vfill
    \begin{minted}[linenos]{rust}
fn area(shape: Shape) -> f64 {
    match shape {
        Shape::Circle(radius) => radius * radius * 3.14159,
        Shape::Rectangle { width, height } => {
            let area = width * height;

            area
        }
        Shape::Triangle { base: b, height: h} => 0.5 * b * h,
    }
}
    \end{minted}
\end{frame}

\begin{frame}[fragile]
    \frametitle{\texttt{match} Expressions}
    \begin{itemize}
        \item Each case of a \texttt{match} expression is called a \textit{match arm}.
        \item Each arm has to evaluate to the same type.
        \pause
        \vfill
        \vspace{.5cm}
        \item We have observed arms of the form \texttt{<pattern> => <expression>}.
        \item Arms can also have the form \texttt{<pattern> <guard> => <expression>}.
        \pause
        \vfill
        \vspace{.5cm}
        \item Usually, an arm is terminated by a \texttt{,}
        \item For the last arm, the \texttt{,} is optional but idiomatic.
        \item If the expression is a block, the \texttt{,} is optional. Skipping it is idiomatic.
        \pause
        \vfill
        \vspace{.5cm}
        \item Examples!
    \end{itemize}
\end{frame}

\begin{frame}[fragile]
    \frametitle{\texttt{match} Expressions}

    \begin{minted}[linenos]{rust}
// Arbitraty score: Two points for large triangles, one for other
// triangles, none for other shapes
fn score(shape: Shape, min_base: f64) -> u8 {
    match shape {
        Shape::Triangle { base, height: _ } if base >= min_base => 2,
        Shape::Triangle { base: _, height: _ } => 1,
        _ => 0,
    }
}
    \end{minted}
\end{frame}

\begin{frame}[fragile]
    \frametitle{\texttt{match} Expressions}

    \begin{minted}[linenos]{rust}
fn xor(arg1: bool, arg2: bool) -> bool {
    match (arg1, arg2) {
        (true, true) => false,
        (true, false) => true,
        (false, true) => true,
        (false, false) => false,
    }
}
    \end{minted}
    \pause
    \vfill
    \begin{minted}[linenos]{rust}
fn xor(arg1: bool, arg2: bool) -> bool {
    match (arg1, arg2) {
        (a, b) if a != b => true,
        _ => false,
    }
}
    \end{minted}
\end{frame}

// return number of iterations it took to reach 1
fn collatz(start: u64) -> u64 {
    let mut i = start;
    let mut steps = 0;
    loop {
        i = match i {
            1 => return steps,
            i if i % 2 == 0 => i / 2,
            i => 3 * i + 1,
        };
        steps += 1;
    }
}
